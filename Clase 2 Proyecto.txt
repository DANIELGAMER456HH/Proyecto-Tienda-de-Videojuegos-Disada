import 'package:flutter/material.dart';

class LoginPage extends StatelessWidget {
  // Controladores para capturar lo que el usuario escribe en el Login
  final TextEditingController userCtrl = TextEditingController();
  final TextEditingController passCtrl = TextEditingController();

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      body: Center( // Centra todo el contenido
        child: Container(
          width: 400, // Ancho fijo para que parezca una tarjeta de login
          padding: EdgeInsets.all(30),
          decoration: BoxDecoration(
            color: Colors.white,
            borderRadius: BorderRadius.circular(20) // Bordes redondeados
          ),
          child: Column(
            mainAxisSize: MainAxisSize.min, // La tarjeta se ajusta al contenido
            children: [
              Icon(Icons.videogame_asset, size: 80, color: Colors.blue), // Icono de control
              Text("VIDEOJUEGOS DISADA", style: TextStyle(fontSize: 24, fontWeight: FontWeight.bold)),
              SizedBox(height: 20),
              TextField(controller: userCtrl, decoration: InputDecoration(labelText: "Usuario Gamer")),
              TextField(controller: passCtrl, obscureText: true, decoration: InputDecoration(labelText: "Contraseña")),
              SizedBox(height: 30),
              ElevatedButton(
                onPressed: () {
                  // Navega a la siguiente pantalla (Formulario)
                  Navigator.pushNamed(context, '/form');
                },
                child: Text("ENTRAR AL SISTEMA"),
              )
            ],
          ),
        ),
      ),
    );
  }
}
